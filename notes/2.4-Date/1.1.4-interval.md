# Interval

都是属于 window 的方法

> 1000毫秒 = 1秒

#### setInterval();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 调用函数或在指定的时间间隔计算表达式（以毫秒为单位）
 * 
 * 
 * handler: 将要执行的 function(){}
 * 
 * timeout?: 执行代码的间隔时间（以毫秒为单位）。如果该值小于10，则使用值10
 * 
 * ...arguments: 传递给函数的其他参数（IE9和更早版本不支持）
 * 
 * 
 * return: 一个数字，表示被设置的定时器的ID值。使用此值与调用 clearInterval(); 方法来取消定时器
 */

setInterval(handler: TimerHandler, timeout?: number, ...arguments: any[]): number;


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * 测试时间计时器是否准确
 * 
 * 
 * 属于异步操作
 * 
 * 非常不准: 执行队列争抢时间片，代码执行损耗可以忽略不计
 */

var fitstTime = new Date().getTime();

setInterval(function(

    var lastTime = new Date().getTime();
    
    console.log(lastTime - firsttime);
    
    firstTime = lastTime;
    
){});


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


let timer;

function interval(callback, duration){

    timer = setTimeout(function(){
    
        callback();
        
        interval(callback, duration);

    }, duration)
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

/**

#### setTimeout();

> 只执行一次的功能。如果你需要重复执行，使用 setInterval();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 调用函数或毫秒的指定次数后计算表达式
 * 
 * 
 * handler: 将要执行的 function(){}
 * 
 * timeout?: 执行代码的间隔时间（以毫秒为单位）。如果该值小于10，则使用值10
 * 
 * ...arguments: 传递给函数的其他参数（IE9和更早版本不支持）
 * 
 * 
 * return: 一个数字，表示被设置的定时器的ID值。使用此值与 clearTimeout(); 方法来取消定时器
 */

setTimeout(handler: TimerHandler, timeout?: number, ...arguments: any[]): number;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
