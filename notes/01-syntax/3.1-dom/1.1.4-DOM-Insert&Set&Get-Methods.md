#### innerText

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// textContent 返回的文本内容的所有元素，而 innerText 返回的元素的所有内容

/**
 * 设置或返回指定节点的文本内容( 及其所有后代 )
 * 
 * 输入覆盖: 设置的innerText属性，子节点将被删除 并包含指定字符串的单个Text节点 被替换
 * 
 * 火狐不兼容
 */

innerText: string;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### textContent

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// textContent 返回的文本内容的所有元素，而 innerText 返回的元素的所有内容

/**
 * 设置或返回指定节点的文本内容( 及其所有后代 )
 * 
 * 输入覆盖: 设置的 textContent 属性，子节点将被删除 并包含指定字符串的单个Text节点 被替换
 * 
 * 老版本 IE 不兼容
 */
 
textContent: string | null;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
