#### clipboard

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * e.clipboard || window.clipboardData
 * 
 * 剪切板
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * dom.getData(MIME: string)
 * 
 * @MIME: 数据类型
 * 
 * 在剪切板中获取指定类型的文件
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * dom.setData(MIME: string, value)
 * 
 * 覆盖剪切板数据
 */

//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * clearData();
 * 
 * 清空剪切板数据
 */


//-------------------------------------------------------------------------------------------------------------------//


/**
 * 数字输入框
 * 
 * 不能粘贴除数字以外的
 */


input.onpase = function(e) {

    let text = e.clipboardData.getData('text/plain');
    
    if(!text || !/\d+/.test(text)){
    
        e.preventDefault();
    }
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
