/**
 * 混合对象

 * 
 *

function minin(obj1, obj2) {

    let obj = {};

    for (const prop in obj2) {

        obj[prop] = obj2[prop];
    }

    for (const prop in obj1) {

        if (!(prop in obj2)) {

            obj[prop] = obj1[prop];
        }
    }

    return obj;
}
